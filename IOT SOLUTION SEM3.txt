Q.2 Write a program to measure the distance using ultrasonic sensor and make LED blink using Arduino.

const int trigPin = 9;
const int echoPin = 10;
const int ledPin  = 13;

long duration;
float distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2; // in cm

  if (distance < 20) {
    digitalWrite(ledPin, HIGH);
  } else {
    digitalWrite(ledPin, LOW);
  }

  Serial.print("Distance: ");
  Serial.println(distance);
  
  delay(200);
}

Q. 2 Write a program to detects the vibration of an object with sensor using Arduino.

// Interrupt-based vibration detection (with SPST switch in Proteus)

const byte vibPin = 2;        // interrupt pin (INT0)
const byte ledPin = 13;       // LED indicator

volatile bool vibDetected = false;
volatile unsigned long lastTriggerTime = 0;
const unsigned long debounceDelay = 80; // ms to prevent double triggers

// Interrupt Service Routine
void vibrationISR() {
  unsigned long current = millis();

  // Debounce: ignore triggers happening too close
  if (current - lastTriggerTime > debounceDelay) {
    vibDetected = true;
    lastTriggerTime = current;
  }
}

void setup() {
  pinMode(vibPin, INPUT);   // if needed: INPUT_PULLUP
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);

  // Attach interrupt on rising edge (switch ON = vibration)
  attachInterrupt(digitalPinToInterrupt(vibPin), vibrationISR, RISING);

  Serial.println("Vibration detector ready...");
}

void loop() {
  if (vibDetected) {
    vibDetected = false;

    Serial.println("VIBRATION DETECTED!");

    // Blink LED quickly to show interrupt fired
    for (int i = 0; i < 3; i++) {
      digitalWrite(ledPin, HIGH);
      delay(80);
      digitalWrite(ledPin, LOW);
      delay(80);
    }
  }
}

Q.2 Write a program to sense a finger when it is placed on the board Arduino.

// Finger-touch (simulated) using interrupt on D2 (UNO)
// Works with SW-SPST between +5V and D2 and a 10k pull-down to GND.

const byte touchPin = 2;   // INT0 on UNO
const byte ledPin   = 13;

volatile bool touched = false;
volatile unsigned long lastInterrupt = 0;
const unsigned long debounceMs = 80; // debounce time

// Plain ISR for AVR (no IRAM_ATTR)
void touchISR() {
  unsigned long now = millis();
  if (now - lastInterrupt > debounceMs) {
    touched = true;
    lastInterrupt = now;
  }
}

void setup() {
  pinMode(touchPin, INPUT);    // using external 10k pull-down resistor
  pinMode(ledPin, OUTPUT);
  attachInterrupt(digitalPinToInterrupt(touchPin), touchISR, RISING);
  Serial.begin(9600);
  Serial.println("Ready");
}

void loop() {
  if (touched) {
    touched = false;
    Serial.println("Touch detected!");
    digitalWrite(ledPin, HIGH);
    delay(200);
    digitalWrite(ledPin, LOW);
  }
}

Q.2 Write a program to get temperature notification using Arduino.

#include <DHT.h>

#define DHTPIN 2       // DHT11 data pin connected to D2
#define DHTTYPE DHT11  // Sensor type

DHT dht(DHTPIN, DHTTYPE);

const int ledPin = 13;     // LED pin
const float threshold = 30.0; // Temperature limit in °C

void setup() {
  Serial.begin(9600);
  pinMode(ledPin, OUTPUT);
  dht.begin();
}

void loop() {
  float temp = dht.readTemperature(); // read temp in Celsius

  if (isnan(temp)) {
    Serial.println("Sensor Error! Check wiring...");
    return;
  }

  Serial.print("Temperature: ");
  Serial.print(temp);
  Serial.println(" °C");

  // Notification logic
  if (temp >= threshold) {
    digitalWrite(ledPin, HIGH);   // LED turns ON when too hot
    Serial.println("⚠ HIGH Temperature Warning!");
  } else {
    digitalWrite(ledPin, LOW);    // LED OFF
  }

  delay(1500); // read every 1.5 seconds
}

Q.2 Write a program for LDR to vary the light intensity of LED using Arduino.

const int ldrPin = A0;
const int ledPin = 9;

void setup() {
  Serial.begin(9600);
  pinMode(ledPin, OUTPUT);
}

void loop() {

  int ldrValue = analogRead(ldrPin);   // 0 to 1023
  int brightness = map(ldrValue, 0, 1023, 255, 0);

  analogWrite(ledPin, brightness);

  Serial.print("LDR: ");
  Serial.print(ldrValue);
  Serial.print("  LED Brightness: ");
  Serial.println(brightness);

  delay(50);
}

#Q.2 (b) Get input from two switches and switch on corresponding LEDs.
#sudo apt-get update
#sudo apt-get install python3-rpi.gpio

import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)

LED1 = 17
LED2 = 27
SW1 = 22
SW2 = 23

GPIO.setup(LED1, GPIO.OUT)
GPIO.setup(LED2, GPIO.OUT)
GPIO.setup(SW1, GPIO.IN, pull_up_down=GPIO.PUD_DOWN)
GPIO.setup(SW2, GPIO.IN, pull_up_down=GPIO.PUD_DOWN)

print("Press switches...")

try:
    while True:
        if GPIO.input(SW1) == GPIO.HIGH:
            GPIO.output(LED1, GPIO.HIGH)
        else:
            GPIO.output(LED1, GPIO.LOW)

        if GPIO.input(SW2) == GPIO.HIGH:
            GPIO.output(LED2, GPIO.HIGH)
        else:
            GPIO.output(LED2, GPIO.LOW)

        time.sleep(0.1)

except KeyboardInterrupt:
    GPIO.cleanup()

#sudo python3 s11q2.py

#Q.2 (c) Flash an LED at given ON and OFF time taken from a file.
#sudo apt-get update

import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
LED_PIN = 17
GPIO.setup(LED_PIN, GPIO.OUT)

# Read ON and OFF times from file
with open("times.txt", "r") as f:
    on_time = float(f.readline().strip())
    off_time = float(f.readline().strip())

print(f"Flashing LED: ON={on_time}s, OFF={off_time}s")

try:
    while True:
        GPIO.output(LED_PIN, GPIO.HIGH)
        time.sleep(on_time)
        GPIO.output(LED_PIN, GPIO.LOW)
        time.sleep(off_time)

except KeyboardInterrupt:
    GPIO.cleanup()

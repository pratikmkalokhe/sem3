//Q.1 Write a Java Program to implement Singleton pattern for multithreading. user input. name as s2q1.

import java.util.Scanner;

// Singleton Class
class Singleton {
    private static volatile Singleton instance = null;   // volatile for thread safety
    public String message;

    // private constructor
    private Singleton() {
        message = "Default Message";
    }

    // Thread-safe Singleton instance using Double-Check Locking
    public static Singleton getInstance() {
        if (instance == null) {                 // first check
            synchronized (Singleton.class) {
                if (instance == null) {         // second check
                    instance = new Singleton();
                }
            }
        }
        return instance;
    }
}

// Thread class to print the Singleton message
class MyThread extends Thread {
    public void run() {
        Singleton s = Singleton.getInstance();
        System.out.println("Thread: " + Thread.currentThread().getName() +
                " | Message: " + s.message);
    }
}

public class s2q1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // User input
        System.out.print("Enter a message for Singleton: ");
        String msg = sc.nextLine();

        // Set message in Singleton instance
        Singleton s = Singleton.getInstance();
        s.message = msg;

        // Create multiple threads
        Thread t1 = new MyThread();
        Thread t2 = new MyThread();
        Thread t3 = new MyThread();

        // Start threads
        t1.start();
        t2.start();
        t3.start();
    }
}

import java.util.Observable;
import java.util.Observer;
import java.util.Scanner;

// WeatherData class (Observable)
class WeatherData extends Observable {
    private float temperature;
    private float humidity;
    private float pressure;

    public float getTemperature() {
        return temperature;
    }

    public float getHumidity() {
        return humidity;
    }

    public float getPressure() {
        return pressure;
    }

    public void measurementsChanged() {
        setChanged();       // marks Observable as changed
        notifyObservers();  // notify all observers
    }

    public void setMeasurement(float temperature, float humidity, float pressure) {
        this.temperature = temperature;
        this.humidity = humidity;
        this.pressure = pressure;
        measurementsChanged();
    }
}

// Display class (Observer)

    @Override
    public void update(Observable o, Object arg) {
        if (o instanceof WeatherData) {
            WeatherData wd = (WeatherData) o;
            System.out.println("\n--- Weather Update ---");
            System.out.println("Temperature: " + wd.getTemperature());
            System.out.println("Humidity: " + wd.getHumidity());
            System.out.println("Pressure: " + wd.getPressure());
        }
    }
}

public class s3q1 {

        WeatherData weatherData = new WeatherData();

import java.util.Scanner;

// -------- Command Interface --------
interface Command {
    void execute();
    void undo();
}

// -------- Receiver: Ceiling Fan --------
class CeilingFan {
    private String state = "OFF";

    public void high() {
        state = "HIGH";
        System.out.println("Ceiling Fan is HIGH");
    }

    public void off() {
        state = "OFF";
        System.out.println("Ceiling Fan is OFF");
    }

    public String getState() {
        return state;
    }
}

// -------- Commands --------
class FanHighCommand implements Command {
    CeilingFan fan;
    String prevState;

    public FanHighCommand(CeilingFan fan) {
        this.fan = fan;
    }

    public void execute() {
        prevState = fan.getState();
        fan.high();
    }

    public void undo() {
        if (prevState.equals("OFF"))
            fan.off();
    }
}

class FanOffCommand implements Command {
    CeilingFan fan;
    String prevState;

    public FanOffCommand(CeilingFan fan) {
        this.fan = fan;
    }

    public void execute() {
        prevState = fan.getState();
        fan.off();
    }

    public void undo() {
        if (prevState.equals("HIGH"))
            fan.high();
    }
}

// -------- Invoker --------
class RemoteControl {
    Command slot;
    Command lastCommand;

    public void setCommand(Command command) {
        slot = command;
    }

    public void pressButton() {
        slot.execute();
        lastCommand = slot;
    }

    public void undoButton() {
        System.out.println("Undoing last action...");
        lastCommand.undo();
    }
}

// -------- MAIN --------
public class s7q1 {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        CeilingFan fan = new CeilingFan();

        Command fanHigh = new FanHighCommand(fan);
        Command fanOff = new FanOffCommand(fan);

        RemoteControl remote = new RemoteControl();

        System.out.println("=== Ceiling Fan Remote (UNDO Command Pattern) ===");
        System.out.println("1. Fan HIGH");
        System.out.println("2. Fan OFF");
        System.out.println("3. UNDO last command");
        System.out.print("Enter your choice: ");

        int choice = sc.nextInt();

        switch(choice) {
            case 1: remote.setCommand(fanHigh); break;
            case 2: remote.setCommand(fanOff); break;
            case 3:
                remote.undoButton();
                return;
            default:
                System.out.println("Invalid choice");
                return;
        }

        System.out.println("\n--- OUTPUT ---");
        remote.pressButton();
    }
}

// -------- Target Interface (what client expects) --------
interface BeatModel {
    int getBeat();
}

// -------- Adaptee (existing class) --------
class HeartModel {
    private int heartRate;

    public HeartModel(int heartRate) {
        this.heartRate = heartRate;
    }

    public int getHeartRate() {

    }

        // Adaptee object
        HeartModel heartModel = new HeartModel(rate);

        // Adapter converting it to BeatModel
        BeatModel beatModel = new HeartToBeatAdapter(heartModel);

        System.out.println("\n---- OUTPUT ----");
        System.out.println("Heart Rate (through Adapter as Beat): " + beatModel.getBeat() + " BPM");

    public Volt(int volts) {
        this.volts = volts;

    public int getVolts() {
        return volts;
    }
}

// -------- Socket (Adaptee) --------
class Socket {
    public Volt getVolt() {
        return new Volt(120);   // default 120V
    }
}

// -------- Class Adapter (using inheritance) --------
class SocketAdapter extends Socket {

    public Volt get120Volt() {
        return getVolt();  // from Socket
    }

    public Volt get12Volt() {
        Volt v = getVolt();
        return convertVolt(v, 10); // 120/10 = 12
    }

    public Volt get3Volt() {
        Volt v = getVolt();
        return convertVolt(v, 40); // 120/40 = 3

// -------- MAIN --------
public class s20q1 {
    public static void main(String[] args) throws Exception {

        System.out.println("Enter text (Uppercase will be converted to lowercase):");

        // Read input from user using decorator
        LowerCaseReader lcr = new LowerCaseReader(new InputStreamReader(System.in));

        System.out.println("\n--- Output (Lowercase) ---");

        int c;
        while ((c = lcr.read()) != -1) {   // stop when user presses Ctrl+Z (Windows) / Ctrl+D (Linux)
            System.out.print((char) c);
        }
    }
}
